 15    def hex2RGB(cls, hex):
 16        """ "#FFFFFF" -> [255,255,255]"""
 17        # Pass 16 to the integer function for change of base
 18        return [int(hex[i : i + 2], 16) for i in range(1, 6, 2)]

 20    @classmethod
 21    def RGB2hex(cls, RGB):
 22        '''[255,255,255] -> "#FFFFFF"'''
 23        # Components need to be integers for hex to make sense
 24        RGB = [int(x) for x in RGB]
 25        return "#" + "".join(
 26            ["0{0:x}".format(v) if v < 16 else "{0:x}".format(v) for v in RGB]
 27        )

 29    @classmethod
 30    def __color_dict(cls, gradient):
 31        """Takes in a list of RGB sub-lists and returns dictionary of
 32        colors in RGB and hex form for use in a graphing function
 33        defined later on"""
 34        return {
 35            "hex": [cls.RGB2hex(RGB) for RGB in gradient],
 36            "r": [RGB[0] for RGB in gradient],
 37            "g": [RGB[1] for RGB in gradient],
 38            "b": [RGB[2] for RGB in gradient],
 39        }

 41    @classmethod
 42    def linear_gradient(cls, start_hex, finish_hex="#FFFFFF", n=10):
 43        """returns a gradient list of (n) colors between
 44        two hex colors. start_hex and finish_hex
 45        should be the full six-digit color string,
 46        inlcuding the number sign ("#FFFFFF")"""
 47        # Starting and ending colors in RGB form
 48        if not cls.__valhex(start_hex) and not cls.__valhex(finish_hex):
 49            raise Exception("Invalid color format")
 50        s = cls.hex2RGB(start_hex)
 51        f = cls.hex2RGB(finish_hex)
 52        # Initilize a list of the output colors with the starting color
 53        RGB_list = [s]
 54        # Calcuate a color at each evenly spaced value of t from 1 to n
 55        for t in range(1, n):
 56            # Interpolate RGB vector for color at the current value of t
 57            curr_vector = [
 58                int(s[j] + (float(t) / (n - 1)) * (f[j] - s[j])) for j in range(3)
 59            ]
 60            # Add it to our list of output colors
 61            RGB_list.append(curr_vector)
 62        return cls.__color_dict(RGB_list)
 63
 64    @classmethod
 65    def polylinear_gradient(cls, colors, n):
 66        """returns a list of colors forming linear gradients between
 67        all sequential pairs of colors. "n" specifies the total
 68        number of desired output colors"""
 69        # The number of colors per individual linear gradient
 70        n_out = int(float(n) / (len(colors) - 1))
 71        # returns dictionary defined by color_dict()
 72        gradient_dict = cls.linear_gradient(colors[0], colors[1], n_out)
 73
 74        if len(colors) > 1:
 75            for c1, c2 in zip(colors, colors[1:]):
 76                next = cls.linear_gradient(c1, c2, n_out)
 77                for k in ("hex", "r", "g", "b"):
 78                    # Exclude first point to avoid duplicates
 79                    gradient_dict[k] += next[k][1:]
 80
 81        return gradient_dict

107    @classmethod
108    def labels_colors_map(cls, labels, opacity=0.25):
109        n = len(labels)
110        colors_dict = cls.rainbow(n)
111        R = np.asarray(colors_dict["r"], dtype=np.float) / 255.0
112        G = np.asarray(colors_dict["g"], dtype=np.float) / 255.0
113        B = np.asarray(colors_dict["b"], dtype=np.float) / 255.0
114        A = [opacity] * n
115        palette = list(zip(B.tolist(), G.tolist(), R.tolist(), A))
116        colors = {label: palette[i] for i, label in enumerate(labels)}
117        return colors

177            )

180        ax.set_ylabel("Green Value")
181        ax.set_zlabel("Blue Value")

183        plt.ylim(0, 255)
184        plt.xlim(0, 255)
185
186        # Save two views of each plot

193        plt.show()
