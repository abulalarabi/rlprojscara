212class Arm(object):
213    """A class to represent an arm, based on our implementation a single arm could
214    have multiple links"""
215
216    def __init__(
217        self, origin: Point2D, env_size: Size2D, link_width: int = 1, goal=typing.List
218    ):
219        """
220        :param origin: the origin of the arm
221        :param env_size: the size of the environment
222        :param link_width: the width of the arm link
223        :param goal: the goal point
224        """
225        # arm
226        self.origin = origin
227        self.links = []
228        self.link_width = link_width
229        # goal
230        self.goal_len = 30
231        self.goal = goal if goal else [500, 500, self.goal_len]
232        self.on_goal = 0
233        # env attributes
234        self.state_dim = 9
235        self.action_dim = 2
236        self.action_bound = [-1, +1]
237        self.env_size = env_size
238
239        self.step_size = 0.05  # granularity
240
241    def head(self):
242        """returns the last link of the arm"""
243        return self.links[-1] if len(self.links) > 0 else None
244
245    def tail(self):
246        """returns the first link of the arm"""
247        return self.links[0] if len(self.links) > 0 else None
248    
249    
250    def add_link(self, length: int, color: tuple = (255, 255, 255)):
251        """
252        Adds a link to the arm.
253        :param length:
254        :param width:
255        :param color:
256        :return:
257        """
258        if len(self.links) == 0:
259            self.links.append(
260                ArmLink(length, self.link_width, color, origin=self.origin)
261            )
262        else:
263            self.links.append(
264                ArmLink(length, self.link_width, color, parent=self.links[-1])
265            )
266
267        self.action_dim = len(self.links)
268        self.state_dim = 4 * self.action_dim + 1  # total number of observations
269
270    def draw(self):
271        """
272        Draws the arm.
273        :return:
274        """
275        for link in self.links:
276            link.draw()
277
278    def set_angles(self, *angles: int):
279        """
280        Sets the angles of the arm links.
281        :param angles:
282        :return:
283        """
284        if len(angles) != len(self.links):
285            raise ValueError("Invalid number of angles specified.")
286        for i, angle in enumerate(angles):
287            self[i].angle = deg2rad(angle)
288
289    def get_observation(self, goal: typing.List = None) -> typing.List:
290        """
291        Returns the observation of the arm.
292        :param goal: the goal point
293        :return: the observation of the arm
294        """
295        observation = []
296        for link in self.links:
297            observation.append(link.endpoint.x / self.env_size.width)  # normalize
298            observation.append(link.endpoint.y / self.env_size.height)  # normalize
299
300        for link in self.links:
301            observation.append(
302                (goal[0] - link.endpoint.x) / self.env_size.width
303            )  # normalize
304            observation.append(
305                (goal[1] - link.endpoint.y) / self.env_size.height
306            )  # normalize
307
308        return observation
309
310    def get_reward(self, goal: typing.List) -> float:
311        """
312        Returns the reward of the arm.
313        """
314        return -self.head().distance_to(goal) / max(
315            self.env_size.width, self.env_size.height
316        )
317
318    def step(self, action: typing.List) -> typing.Tuple:
319        """
320        Performs a step in the environment.
321        :param action: the action to perform defined as the angle of each link
322        """
323        done = False
324        for i in range(len(action)):
325            self.links[i].angle += np.clip(action[i], -1, 1) * self.step_size
326
327        r = self.get_reward(self.goal)
328
329        # done and reward
330        if (
331            self.goal[0] - self.goal[2] / 2
332            < self.head().endpoint.x
333            < self.goal[0] + self.goal[2] / 2
334        ):
335            if (
336                self.goal[1] - self.goal[2] / 2
337                < self.head().endpoint.y
338                < self.goal[1] + self.goal[2] / 2
339            ):
340                r += 1.0
341                self.on_goal += 1
342                if self.on_goal > 50:  # if it is over the goal for 50 times
343                    done = True
344        else:
345            self.on_goal = 0
346
347        observations = self.get_observation(self.goal)
348        s = np.concatenate((observations, [1.0 if self.on_goal else 0.0]))
349
350        return s, r, done
351
352    def reset(self):
353        # set the goal approximately withing the arm's range
354        while 1:
355            self.goal = [
356                np.random.rand() * self.env_size.width,
357                np.random.rand() * self.env_size.height,
358                self.goal_len,
359            ]
360            if (
361                math.sqrt(
362                    (self.goal[0] - self.origin.x) ** 2
363                    + (self.goal[1] - self.origin.y) ** 2
364                )
365                > self.links[0].length
366            ):
367                break
368
369        # check if on goal
370        if (
371            self.goal[0] - self.goal[2] / 2
372            < self.head().endpoint.x
373            < self.goal[0] + self.goal[2] / 2
374        ):
375            if (
376                self.goal[1] - self.goal[2] / 2
377                < self.head().endpoint.y
378                < self.goal[1] + self.goal[2] / 2
379            ):
380                self.on_goal = 1
381        else:
382            self.on_goal = 0
383
384        for link in self.links:  # randomize arm angles
385            link.angle = math.pi * np.random.rand(1)[0]
386
387        observations = self.get_observation(self.goal)
388
389        s = np.concatenate((observations, [1.0 if self.on_goal else 0.0]))
390        return s

392    def setenv(self, goal) -> np.ndarray:
393
394        self.goal = goal
395        # check if on goal
396        if (
397            self.goal[0] - self.goal[2] / 2

410        # for link in self.links: # randomize arm angles

419        return self.links[item] if item < len(self.links) else None
